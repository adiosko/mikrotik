class IPv6Route:

    def listROutes(self): 
        """ 
        Method will list routes 
        :return: 
        """ 
        ipv6 = self.client.talk(['/ipv6/route/print']) 
        if ipv6 == {}: 
            print("No route found") 
        else: 
            print("Dst address\tGateway\tDisatnce") 
            for i in ipv6: 
                print(ipv6[i]['dst-address']+"\t"+ipv6[i]['gateway']+"\t"+ipv6[i]['distance']) 
        return ipv6
