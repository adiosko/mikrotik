# Shim to wrap setup.py invocation with setuptools 
SETUPTOOLS_SHIM = ( 
    "import setuptools, tokenize;__file__=%r;" 
    "f=getattr(tokenize, 'open', open)(__file__);" 
    "code=f.read().replace('\\r\\n', '\\n');" 
    "f.close();" 
    "exec(compile(code, __file__, 'exec'))" 
)
